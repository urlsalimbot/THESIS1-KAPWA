    
    Update the implementation:

    1. **Barangay filter** (existing, unchanged): `b.address ILIKE :barangay`
    
    2. **Category filter** (new — added BEFORE search to be independent):
       ```typescript
       if (category) {
         qb.andWhere('b.category = :category', { category });
       }
       ```
       Also search the category field when a general search term is provided:
       ```typescript
       OR b.category ILIKE :categoryMatch
       ```
       (Only when search is active and search.length >= 2)
    
    3. **Search enhancement** (replace existing plainto_tsquery + ILIKE):
       ```typescript
       if (search && search.length >= 2) {
         qb.andWhere(`
           (b.search_vector @@ plainto_tsquery('english', :search)
            OR similarity(b.surname, :search) > 0.3
            OR similarity(b.first_name, :search) > 0.3
            OR b.category ILIKE :categoryMatch)
         `, { search, categoryMatch: `%${search}%` });
         
         qb.addSelect(`
           COALESCE(ts_rank(b.search_vector, plainto_tsquery('english', :search2)), 0) + 
           COALESCE(similarity(b.surname, :search2), 0) + 
           COALESCE(similarity(b.first_name, :search2), 0)
         `, 'rank')
         .orderBy('rank', 'DESC');
         
         qb.setParameters({ search, search2: search, categoryMatch: `%${search}%` });
       }
       ```
    
    4. **Short query guard**: For search.length < 3, use tsvector + ILIKE fallback only (skip trigram similarity to avoid false positives per RESEARCH.md Pitfall 3):
       ```typescript
       if (search && search.length >= 2) {
         if (search.length >= 3) {
           // Full trigram + tsvector path (as above)
         } else {
           // Short query: tsvector + ILIKE only (no trigram)
           qb.andWhere(
             `(b.search_vector @@ plainto_tsquery('english', :search)
               OR b.surname ILIKE :like
               OR b.first_name ILIKE :like)`,
             { search, like: `%${search}%` }
           );
           qb.orderBy('ts_rank(b.search_vector, plainto_tsquery(:search))', 'DESC');
         }
       }
       ```
