async getOrCreateTrackerId(date: Date, clientSeqNum?: number): Promise<{ trackerId: string; seqNum: number }> {
  // If client provides a seqNum, try using it; if conflict, fall back to server-generated
  if (clientSeqNum) {
    const existing = await this.trackerRepo.findOne({
      where: { transactionDate: Between(...this.getDayRange(date)), dailySeqNum: clientSeqNum }
    });
    if (existing) {
      // Use existing entry — resolve conflict per D-08: server validates/resolves
      return { trackerId: existing.trackerId || await this.generateTrackerId(date, existing.dailySeqNum), seqNum: existing.dailySeqNum };
    }
  }
  const seq = clientSeqNum || (await this.getNextSequence(date));
  const trackerId = await this.generateTrackerId(date, seq);
  return { trackerId, seqNum: seq };
}
